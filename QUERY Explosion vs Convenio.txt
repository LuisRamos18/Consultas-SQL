SELECT T0.OBRA, T0.INSUMO,T1.DESCRIPCION, T1.UNIDAD, CANTIDAD_BASE * 1 AS CANTIDAD, PRECIO_BASE * 1 AS PRECIO, IMPORTE_BASE * 1 AS IMPORTE,
T2.CANTIDAD_CONTRATO, T2.PRECIO_CONTRATO,T2.STATUS, T2.TIPO
FROM  dba.SU_EXPLOSION_BASE AS T0
INNER JOIN dba.insumos AS T1 ON T0.INSUMO = T1.INSUMO
LEFT JOIN dba.su_contrato_contratista_det AS T2 ON T0.INSUMO = T2.INSUMO AND T0.OBRA = T2.OBRA
WHERE T0.OBRA = '2H3' AND T0.INSUMO BETWEEN '2010001' AND '2980009'
ORDER BY T0.INSUMO
